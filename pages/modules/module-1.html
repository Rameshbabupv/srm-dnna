üíª Python Implementation: AND Gate Perceptron
import numpy as np

class Perceptron:
    def __init__(self, learning_rate=0.1, epochs=100):
        self.learning_rate = learning_rate
        self.epochs = epochs
    
    def fit(self, X, y):
        self.weights = np.zeros(X.shape[1])
        self.bias = 0
        
        for epoch in range(self.epochs):
            for i in range(X.shape[0]):
                prediction = self.predict_single(X[i])
                error = y[i] - prediction
                
                # Update weights and bias
                self.weights += self.learning_rate * error * X[i]
                self.bias += self.learning_rate * error
    
    def predict_single(self, x):
        return 1 if np.dot(x, self.weights) + self.bias >= 0 else 0

# AND Gate Training Data
X_and = np.array([[0, 0], [0, 1], [1, 0], [1, 1]])
y_and = np.array([0, 0, 0, 1])

# Create and train perceptron
and_perceptron = Perceptron()
and_perceptron.fit(X_and, y_and)

üíª TensorFlow Implementation: XOR Solution
import tensorflow as tf
import numpy as np

# XOR dataset
X = np.array([[0, 0], [0, 1], [1, 0], [1, 1]], dtype=np.float32)
y = np.array([[0], [1], [1], [0]], dtype=np.float32)

# Build MLP model
model = tf.keras.Sequential([
    tf.keras.layers.Dense(4, activation='relu', input_shape=(2,)),
    tf.keras.layers.Dense(1, activation='sigmoid')
])

# Compile model
model.compile(optimizer='adam',
              loss='binary_crossentropy',
              metrics=['accuracy'])

# Train model
history = model.fit(X, y, epochs=1000, verbose=0)

# Test predictions
predictions = model.predict(X)
print("XOR Predictions:", predictions.flatten())

üìö Day 3 (Wednesday): Activation Functions & Architecture

‚è∞ Time: 8:00am - 9:40am IST (1h 40m)
üìä Key Concepts
  • Why Activation Functions: Introduce non-linearity
  • Sigmoid: œÉ(x) = 1/(1+e^(-x)), smooth, bounded [0,1]
  • ReLU: f(x) = max(0,x), simple, avoids vanishing gradients
  • Tanh: tanh(x) = (e^x - e^(-x))/(e^x + e^(-x)), zero-centered
  • Softmax: For multi-class classification, probability distribution
üíª Activation Function Comparison
import numpy as np
import matplotlib.pyplot as plt

x = np.linspace(-5, 5, 100)

# Define activation functions
def sigmoid(x):
    return 1 / (1 + np.exp(-x))

def relu(x):
    return np.maximum(0, x)

def tanh(x):
    return np.tanh(x)

# Plot comparison
plt.figure(figsize=(12, 4))

plt.subplot(1, 3, 1)
plt.plot(x, sigmoid(x), label='Sigmoid', color='blue')
plt.title('Sigmoid Activation')
plt.grid(True)

plt.subplot(1, 3, 2)
plt.plot(x, relu(x), label='ReLU', color='red')
plt.title('ReLU Activation')
plt.grid(True)

plt.subplot(1, 3, 3)
plt.plot(x, tanh(x), label='Tanh', color='green')
plt.title('Tanh Activation')
plt.grid(True)

plt.tight_layout()
plt.show()
